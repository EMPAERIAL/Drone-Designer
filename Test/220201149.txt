#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_ARAC 20
#define MAX_DONEM 4



typedef struct {
    char plaka[16];
    char soforAd[31];
    float donemlikYakit[MAX_DONEM];
} AracVerisi;

int veriGirisYap(AracVerisi araclar[]) {
    int n;
    printf("Kac adet arac girilecek?: ");
    scanf("%d", &n);
    getchar();

    for (int i = 0; i < n; i++) {
        printf("\n--- %d. Arac Bilgileri ---\n", i + 1);

        printf("Plaka giriniz : ");
        fgets(araclar[i].plaka, sizeof(araclar[i].plaka), stdin);
        araclar[i].plaka[strcspn(araclar[i].plaka, "\n")] = '\0';

        printf("Sofor Adi Soyadi giriniz : ");
        fgets(araclar[i].soforAd, sizeof(araclar[i].soforAd), stdin);
        araclar[i].soforAd[strcspn(araclar[i].soforAd, "\n")] = '\0';

        printf("4 Donemlik Yakit Tuketimi (Litre):\n");
        for (int j = 0; j < MAX_DONEM; j++) {
            printf("%d. Donem: ", j + 1);
            scanf("%f", &araclar[i].donemlikYakit[j]);
        }
        getchar();
    }
    return n;
}



void maliyetHesapla(AracVerisi arac, float fiyat) {
    float toplam = 0;
    for (int i = 0; i < MAX_DONEM; i++)
        toplam += arac.donemlikYakit[i];

    float maliyet = toplam * fiyat;
    printf("Plaka: %s | Sofor: %s | Yillik Maliyet: %.2f TL | Toplam Tuketim: %.2f Litre\n",
           arac.plaka, arac.soforAd, maliyet, toplam);
}

void limitAsimRaporu(AracVerisi araclar[], int n) {
    printf("\n--- LIMIT ASIM RAPORU ---\n");
    for (int i = 0; i < n; i++) {
        float toplam = 0;
        for (int j = 0; j < MAX_DONEM; j++)
            toplam += araclar[i].donemlikYakit[j];

        if (toplam > 1500) {
            float maks = araclar[i].donemlikYakit[0];
            for (int j = 1; j < MAX_DONEM; j++)
                if (araclar[i].donemlikYakit[j] > maks)
                    maks = araclar[i].donemlikYakit[j];

            printf("Uyari! %s (Sofor: %s) limiti asti.\n", araclar[i].plaka, araclar[i].soforAd);
            printf("En yuksek tuketim su donemlerde goruldu:");
            for (int j = 0; j < MAX_DONEM; j++)
                if (araclar[i].donemlikYakit[j] == maks)
                    printf(" %d. Donem", j + 1);
            printf(" (%.2f Litre)\n", maks);
        }
    }
}





void aracAra(AracVerisi araclar[], int n) {
    int secim;
    char aranan[31];

    while (1) {
        printf("\n--- ARAC ARAMA MENUSU ---\n");
        printf("1- Sofor adina gore ara\n");
        printf("2- Cikis\n");
        printf("Seciminiz: ");
        scanf("%d", &secim);
        getchar();

        if (secim == 2) {
            printf("Arama menusunden cikiliyor. Iyi gunler!\n");
            break;
        }

        if (secim == 1) {
            printf("Aranan Sofor Adini giriniz: ");
            fgets(aranan, sizeof(aranan), stdin);
            aranan[strcspn(aranan, "\n")] = '\0';

            int bulundu = 0;
            for (int i = 0; i < n; i++) {
                if (strcmp(araclar[i].soforAd, aranan) == 0) {
                    printf("\nBulunan Kayit -> Plaka: %s | Sofor: %s\n", araclar[i].plaka, araclar[i].soforAd);
                    printf("Donemlik Yakit Listesi: ");
                    for (int j = 0; j < MAX_DONEM; j++) {
                        printf("%d.D(%.2f)", j + 1, araclar[i].donemlikYakit[j]);
                        if (j < MAX_DONEM - 1) printf(" | ");
                    }
                    printf("\n");
                    bulundu = 1;
                }
            }
            if (!bulundu)
                printf("Kayit bulunamadi.\n");
        }
    }
}



int main() {
    AracVerisi araclar[MAX_ARAC];
    int aracSayisi = veriGirisYap(araclar);

    float fiyat;
    printf("\nLutfen guncel yakit litre fiyatini giriniz (TL): ");
    scanf("%f", &fiyat);

    printf("\n--- YILLIK MALIYET RAPORU ---\n");
    for (int i = 0; i < aracSayisi; i++)
        maliyetHesapla(araclar[i], fiyat);

    limitAsimRaporu(araclar, aracSayisi);
    aracAra(araclar, aracSayisi);

    return 0;
}
